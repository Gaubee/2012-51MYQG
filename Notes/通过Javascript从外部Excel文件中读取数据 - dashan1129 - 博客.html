 2     function f_ReadExcel() {
 3         $("#btnYes")[0].disabled = true;
 4         var filePath = $("#txtFile")[0].value;
 5         if (filePath == "" || filePath == null || filePath == "undefined") {
 6             alert("请选择一个Excel文档");
 7             return;
 8         }
 9         var oXL = new ActiveXObject("Excel.application");
10         var oWB = oXL.Workbooks.open(filePath);
11         oWB.worksheets(1).select();
12         var oSheet = oWB.ActiveSheet;
13         try {
14             var count = $("#count")[0];
15             var i = parseInt(count.value);
16             var j = 2;
17             if (i == 1) {
18                 if ($("#txtEnglishName" + i)[0].value == "") {
19                     --i;
20                 }
21             }
22             while(true)
23             {
24                 if (oSheet.Cells(j, 1).value == undefined || oSheet.Cells(j, 3).value == undefined || oSheet.Cells(j, 6).value == undefined)
25                 {
26                     break;
27                 }
28                 f_AddRow();
29                 i++;
30                 $("#txtLoadingDate" + i)[0].value = f_ToDate(oSheet.Cells(j, 1).value);
31                 $("#txtPONumber" + i)[0].value = oSheet.Cells(j, 2).value == undefined ? "" : oSheet.Cells(j, 2).value;
32                 $("#txtEnglishName" + i)[0].value = oSheet.Cells(j, 3).value;
33                 $("#txtProductName" + i)[0].value = oSheet.Cells(j, 4).value == undefined ? "" : oSheet.Cells(j, 4).value;
34                 $("#txtISBN" + i)[0].value = oSheet.Cells(j, 5).value == undefined ? "" : oSheet.Cells(j, 5).value;
35                 $("#txtQuantity" + i)[0].value = oSheet.Cells(j, 6).value;
36                 $("#txtQtyPerCarton" + i)[0].value = oSheet.Cells(j, 7).value == undefined ? 0 : oSheet.Cells(j, 7).value;
37                 $("#txtCarton" + i)[0].value = oSheet.Cells(j, 8).value == undefined ? 0 : oSheet.Cells(j, 8).value;
38                 $("#txtCtnPerCrate" + i)[0].value = oSheet.Cells(j, 9).value == undefined ? 0 : oSheet.Cells(j, 9).value;
39                 $("#txtCrate" + i)[0].value = oSheet.Cells(j, 10).value == undefined ? 0 : oSheet.Cells(j, 10).value;
40                 $("#txtCW" + i)[0].value = oSheet.Cells(j, 11).value == undefined ? 0 : oSheet.Cells(j, 11).value;
41                 $("#txtCtnSize" + i)[0].value = oSheet.Cells(j, 12).value == undefined ? "" : oSheet.Cells(j, 12).value;
42                 $("#txtProductSize" + i)[0].value = oSheet.Cells(j, 13).value == undefined ? "" : oSheet.Cells(j, 13).value;
43                 $("#txtCrSize" + i)[0].value = oSheet.Cells(j, 14).value == undefined ? "" : oSheet.Cells(j, 14).value;
44                 $("#txtTailSize" + i)[0].value = oSheet.Cells(j, 15).value == undefined ? "" : oSheet.Cells(j, 15).value;

46                 $("#txtNWPerCarton" + i)[0].value = oSheet.Cells(j, 17).value == undefined ? 0 : f_Round(oSheet.Cells(j, 17).value, 2);

49                 $("#txtGWT" + i)[0].value = oSheet.Cells(j, 20).value == undefined ? 0 : f_Round(oSheet.Cells(j, 20).value, 2);
50                 $("#txtUnitPrice" + i)[0].value = oSheet.Cells(j, 21).value == undefined ? 0 : f_Round(oSheet.Cells(j, 21).value, 4);
51                 $("#txtAmount" + i)[0].value = oSheet.Cells(j, 22).value == undefined ? 0 : f_Round(oSheet.Cells(j, 22).value, 2);
52                 $("#txtRemark" + i)[0].value = oSheet.Cells(j, 23).value == undefined ? "" : oSheet.Cells(j, 23).value;
53                 j++;
54             }

57         } catch (e) {

62     }
